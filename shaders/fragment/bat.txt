#version 130

in vec2 iuvs; // Współrzędne teksturowania interpolowane

out vec4 pixelColor; //Zmienna wyjsciowa fragment shadera. Zapisuje sie do niej ostateczny (prawie) kolor piksela
uniform sampler2D textureMap0;

void main(void) {
	vec4 texColor = texture(textureMap0,iuvs);
	
	pixelColor = vec4(texColor.rgb,texColor.a);
}

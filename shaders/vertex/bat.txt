#version 130

//Zmienne jednorodne
uniform mat4 MVP;

//Atrybuty
in vec4 vertex; 
in vec2 texCoord;  
in vec4 normal; 

out vec2 iTexCoord; // Współrzędne tekstur do interpolacji 
out vec4 iVertex;
out vec4 iNormal;


void main(void) {
	iVertex = vertex;
	iNormal = normal;
	iTexCoord = texCoord;
	
	gl_Position = MVP*vertex;
}
